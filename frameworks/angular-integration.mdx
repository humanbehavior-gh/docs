## Step 2: Basic Setup

### Add the Tracker to Your App

**For Angular 17+ (Standalone Components):**

In your `main.ts` file:

```typescript
import { bootstrapApplication } from '@angular/platform-browser';
import { AppComponent } from './app/app.component';
import { initializeHumanBehavior } from 'humanbehavior-js/angular';

// Initialize the tracker once globally
if (typeof window !== 'undefined') {
  initializeHumanBehavior(process.env.HUMANBEHAVIOR_API_KEY || '');
}

bootstrapApplication(AppComponent, {
  providers: [
    // Your other providers
  ]
}).catch(err => console.error(err));
```

**For Legacy Angular (NgModule):**

In your `app.module.ts` file:

```typescript
import { NgModule } from '@angular/core';
import { BrowserModule } from '@angular/platform-browser';
import { AppComponent } from './app.component';
import { HumanBehaviorModule } from 'humanbehavior-js/angular';
import { environment } from '../environments/environment';

@NgModule({
  declarations: [AppComponent],
  imports: [
    BrowserModule,
    HumanBehaviorModule.forRoot({
      apiKey: environment.humanBehaviorApiKey
    })
  ],
  providers: [],
  bootstrap: [AppComponent]
})
export class AppModule { }
```

<Note>
  **That's it!** The SDK now automatically tracks user interactions, navigation, and console events. Everything past here is optional.
</Note>

## Optional: Advanced Features

### Track Custom Events

**For Angular 17+ (Standalone Components):**

```typescript
import { Component, OnInit } from '@angular/core';
import { initializeHumanBehavior } from 'humanbehavior-js/angular';

@Component({
  selector: 'app-my-component',
  template: `
    <div>
      <h1>My Component</h1>
      <button (click)="handleButtonClick()">Click Me</button>
    </div>
  `,
  standalone: true
})
export class MyComponent implements OnInit {
  private tracker = initializeHumanBehavior(process.env.HUMANBEHAVIOR_API_KEY || '');

  ngOnInit() {
    // Track component view
    this.tracker.customEvent('component_viewed', {
      componentName: 'MyComponent'
    });
  }

  async handleButtonClick() {
    await this.tracker.customEvent('button_clicked', {
      buttonLocation: 'header'
    });
  }
}
```

**For Legacy Angular (NgModule):**

```typescript
import { Component, OnInit, Inject } from '@angular/core';
import { HumanBehaviorTracker } from 'humanbehavior-js';

@Component({
  selector: 'app-my-component',
  template: `
    <div>
      <h1>My Component</h1>
      <button (click)="handleButtonClick()">Click Me</button>
    </div>
  `
})
export class MyComponent implements OnInit {
  constructor(@Inject(HumanBehaviorTracker) private tracker: HumanBehaviorTracker) {}

  ngOnInit() {
    // Track component view
    this.tracker.customEvent('component_viewed', {
      componentName: 'MyComponent'
    });
  }

  async handleButtonClick() {
    await this.tracker.customEvent('button_clicked', {
      buttonLocation: 'header'
    });
  }
}
```

### User Authentication Example

**For Angular 17+ (Standalone Components):**

```typescript
import { Component } from '@angular/core';
import { initializeHumanBehavior } from 'humanbehavior-js/angular';

@Component({
  selector: 'app-login-form',
  template: `
    <form (ngSubmit)="handleLogin()">
      <input [(ngModel)]="email" type="email" placeholder="Email" name="email" />
      <input [(ngModel)]="password" type="password" placeholder="Password" name="password" />
      <button type="submit">Login</button>
    </form>
  `,
  standalone: true,
  imports: [FormsModule]
})
export class LoginFormComponent {
  email = '';
  password = '';
  private tracker = initializeHumanBehavior(process.env.HUMANBEHAVIOR_API_KEY || '');

  async handleLogin() {
    try {
      // Your login logic here
      const user = await this.login(this.email, this.password);
      
      if (user) {
        // Identify user with current endUserId
        await this.tracker.identifyUser({
          userProperties: {
            email: user.email,
            name: user.name,
            userId: user.id,
            plan: user.plan
          }
        });
      }
    } catch (error) {
      console.error('Login failed:', error);
    }
  }
}
```

**For Legacy Angular (NgModule):**

```typescript
import { Component, Inject } from '@angular/core';
import { HumanBehaviorTracker } from 'humanbehavior-js';

@Component({
  selector: 'app-login-form',
  template: `
    <form (ngSubmit)="handleLogin()">
      <input [(ngModel)]="email" type="email" placeholder="Email" name="email" />
      <input [(ngModel)]="password" type="password" placeholder="Password" name="password" />
      <button type="submit">Login</button>
    </form>
  `
})
export class LoginFormComponent {
  email = '';
  password = '';

  constructor(@Inject(HumanBehaviorTracker) private tracker: HumanBehaviorTracker) {}

  async handleLogin() {
    try {
      // Your login logic here
      const user = await this.login(this.email, this.password);
      
      if (user) {
        // Identify user with current endUserId
        await this.tracker.identifyUser({
          userProperties: {
            email: user.email,
            name: user.name,
            userId: user.id,
            plan: user.plan
          }
        });
      }
    } catch (error) {
      console.error('Login failed:', error);
    }
  }
}
```

## Environment Variables

```bash
# .env
HUMANBEHAVIOR_API_KEY=your-api-key-here
```

**For Angular 17+ (Standalone Components):**
```typescript
// In your code
const tracker = initializeHumanBehavior(process.env.HUMANBEHAVIOR_API_KEY || '');
```

**For Legacy Angular (NgModule):**
```typescript
// In your code
// Add to src/environments/environment.ts
export const environment = {
  production: false,
  humanBehaviorApiKey: process.env.HUMANBEHAVIOR_API_KEY || ''
};
```

## Troubleshooting

- Check browser console for any error messages
- Verify your API key is correct
- Ensure the tracker is initialized before any user interaction
